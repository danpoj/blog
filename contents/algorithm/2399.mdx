```js title="solution 1"
const fs = require('fs')
const path = process.platform === 'linux' ? '/dev/stdin' : './input.txt'

let [n, arr] = fs.readFileSync(path).toString().trim().split('\n')

arr = arr
  .split(' ')
  .map((s) => +s)
  .sort((a, b) => a - b)

let result = 0n

arr.forEach((num, index) => {
  result += BigInt(num) * BigInt(2 * index - n + 1)
})

console.log(String(result * BigInt(2)))
```
