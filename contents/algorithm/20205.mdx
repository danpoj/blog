<Callout emoji='✔️'>[20205 - 교수님 그림이 깨지는데요?](https://www.acmicpc.net/problem/20205)</Callout>

```js title="Javascript"
const fs = require('fs')
const path = process.platform === 'linux' ? '/dev/stdin' : './input.txt'

let [nk, ...arr] = fs.readFileSync(path).toString().trim().split('\n')

let [n, k] = nk.split(' ').map((c) => +c)

arr = arr.map((s) => s.split(' '))

let result = ''

for (let i = 0; i < n; i++) {
  const newStr = arr[i].map((el) => el.repeat(k)).join('')

  for (let j = 0; j < k; j++) {
    result += newStr.split('').join(' ') + '\n'
  }
}

console.log(result.slice(0, -1))
```
