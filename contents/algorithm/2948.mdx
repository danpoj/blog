```js title="Javascript"
const fs = require('fs')
const path = process.platform === 'linux' ? '/dev/stdin' : './input.txt'

let [day, month] = fs
  .readFileSync(path)
  .toString()
  .trim()
  .split(' ')
  .map((s) => +s)

const days = [31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31]
const cu_days = [0]

const names = ['Thursday', 'Friday', 'Saturday', 'Sunday', 'Monday', 'Tuesday', 'Wednesday']

for (let i = 0; i < days.length; i++) {
  cu_days[i + 1] = cu_days[i] + days[i]
}

let day_diff = cu_days[month - 1] + day - 1

day_diff %= 7

console.log(names[day_diff])
```
