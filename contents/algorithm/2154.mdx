```js title="solution 1"
const fs = require('fs')
const path = process.platform === 'linux' ? '/dev/stdin' : './input.txt'

let target = fs.readFileSync(path).toString().trim()

let str = ''

for (let i = 1; i <= +target; i++) {
  str += String(i)
}

for (let i = 0; i < str.length; i++) {
  if (str.startsWith(target, i)) {
    console.log(i + 1)
    break
  }
}
```
