```js title="solution 1"
const fs = require('fs')
const path = process.platform === 'linux' ? '/dev/stdin' : './input.txt'

let n = +fs.readFileSync(path).toString().trim()

let result = 0

let len = String(n).split('').length

for (let i = n - 9 * len; i <= n; i++) {
  if (i <= 0) continue

  const target =
    i +
    i
      .toString()
      .split('')
      .reduce((prev, cur) => prev + Number(cur), 0)

  if (target === n) {
    result = i
    break
  }
}

console.log(result)
```
