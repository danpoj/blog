```js title="Javascript"
const fs = require('fs')
const path = process.platform === 'linux' ? '/dev/stdin' : './input.txt'

let [n, ...arr] = fs.readFileSync(path).toString().trim().split('\n')

let result = ''

for (const str of arr) {
  if (str.length !== 3) continue

  if (result === '') {
    result = str
    continue
  }

  if (result > str) {
    result = str
  }
}

console.log(result)
```
