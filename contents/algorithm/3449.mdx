```js title="solution"
const fs = require('fs')
const path = process.platform === 'linux' ? '/dev/stdin' : './input.txt'

const [n, ...arr] = fs.readFileSync(path).toString().trim().split('\n')

const result = []

for (let i = 0; i < n * 2; i += 2) {
  const a = arr[i]
  const b = arr[i + 1]
  const len = a.length

  let res = 0

  for (let j = 0; j < len; j++) {
    if (a[j] !== b[j]) {
      res++
    }
  }

  result.push(`Hamming distance is ${res}.`)
}

console.log(result.join('\n'))
```
