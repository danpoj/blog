```js title="Javascript"
const fs = require('fs')
const path = process.platform === 'linux' ? '/dev/stdin' : './input.txt'

let num = +fs.readFileSync(path).toString().trim()

num++

while (true) {
  let str = num.toString()

  let isDistinct = str.length === new Set(str).size

  if (isDistinct) {
    break
  }

  num++
}

console.log(num)
```
